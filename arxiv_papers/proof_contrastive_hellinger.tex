\subsection{Hellinger Distance Guarantee}\label{hellinger_guarantee}
Suppose that $\hat\theta$ is the output of Algorithm \ref{mle+erm}, which satisfies 
\%
\hat\theta\leftarrow \argmax_{\theta\in\Theta}\sum^{m}_{i=1}\log p_{f_{\theta}}(x_i,x'_{i},t_i).
\%
We have the following theoretical guarantee on the Hellinger distance between $\P_{f_{\hat\theta}}(x,x',t)$ and $\P_{f_{\theta^* }}(x,x',t)$.
\begin{lemma}\label{mle_hellinger}
Let $\hat\theta$ be the output of Algorithm \ref{mle+erm}. It then holds that with probability at least $1-\delta$ that
\%
H\big(\P_{f_{\hat\theta}}(x,x',t),\P_{f_{\theta^* }}(x,x',t)\big)\leq \sqrt{\frac{2}{m}\log\frac{N_{\b}\big(\mP(\mathcal{F}_{\theta}),1/m^2\big)}{\delta}},
\%
where we define $\mP(\mathcal{F}_{\theta}):=\{p_{f_{\theta}}(x,x',t)\,|\,\theta\in\Theta\}$.
\end{lemma}
\begin{proof}[Proof of Lemma \ref{mle_hellinger}]
For notation simplicity, we denote $\bm{x}:=(x,x',t)$. Let $\epsilon>0$. Similar to the proof of Theorem \ref{tv_mle}, we obtain with probability at least $1-\delta$
\%\label{101508}
1-\int\sqrt{\bar{p}_{f_{\hat\theta}}(\bm{x})p_{f_{\theta^* }}(\bm{x})}\,d\bm{x}\leq \frac{1}{m}\log\frac{N_{\b}\big(\mP(\mathcal{F}_{\theta}),\epsilon\big)}{\delta}.
\%
Here $\bar{p}_{f_{\hat\theta}}(\bm{x})\in\mN_{\b}(\mP(\mathcal{F}_{\theta}),\epsilon)$ that satisfies $\bar{p}_{f_{\hat\theta}}(\bm{x})\geq p_{f_{\hat\theta}}(\bm{x})$ for any $\bm{x}$ and 
\%
\int \bar{p}_{f_{\hat\theta}}(\bm{x})-p_{f_{\hat\theta}}(\bm{x})\,d\bm{x}\leq \epsilon.
\%
Note that
\%\label{101509}
&1-\int\sqrt{{p}_{f_{\hat\theta}}(\bm{x})p_{f_{\theta^* }}(\bm{x})}\,d\bm{x}-\bigg(1-\int\sqrt{\bar{p}_{f_{\hat\theta}}(\bm{x})p_{f_{\theta^* }}(\bm{x})}\,d\bm{x}\bigg)\notag\\
&=\int\Big(\sqrt{\bar{p}_{f_{\hat\theta}}(\bm{x})}-\sqrt{{p}_{f_{\hat\theta}}(\bm{x})}\Big)\sqrt{{p}_{f_{\theta^* }}(\bm{x})}\,d\bm{x}\notag\\
&\leq \sqrt{\int\Big(\sqrt{\bar{p}_{f_{\hat\theta}}(\bm{x})}-\sqrt{{p}_{f_{\hat\theta}}(\bm{x})}\Big)^2\,d\bm{x}}\notag\\
&=\sqrt{\int \bar{p}_{f_{\hat\theta}}(\bm{x})+{p}_{f_{\hat\theta}}(\bm{x})-2\sqrt{\bar{p}_{f_{\hat\theta}}(\bm{x}){p}_{f_{\hat\theta}}(\bm{x})} \,d\bm{x}}\notag\\
&\leq \sqrt{\int \bar{p}_{f_{\hat\theta}}(\bm{x})-{p}_{f_{\hat\theta}}(\bm{x})\,d\bm{x}}\notag\\
&\leq \sqrt{\epsilon}.
\%
Here the first inequality follows from Cauchy-Schwarz inequality and the second follows from the fact that $\sqrt{\bar{p}_{f_{\hat\theta}}(\bm{x}){p}_{f_{\hat\theta}}(\bm{x})}\geq{p}_{f_{\hat\theta}}(\bm{x})$. By \eqref{101508} and \eqref{101509}, we have
\%
1-\int\sqrt{{p}_{f_{\hat\theta}}(\bm{x})p_{f_{\theta^* }}(\bm{x})}\,d\bm{x}\leq \sqrt{\epsilon}+\frac{1}{m}\log\frac{N_{\b}\big(\mP(\mathcal{F}_{\theta}),\epsilon\big)}{\delta},
\%
which implies that
\%
H^2\big({p}_{f_{\hat\theta}}(\bm{x}),p_{f_{\theta^* }}(\bm{x})\big)=1-\int\sqrt{{p}_{f_{\hat\theta}}(\bm{x})p_{f_{\theta^* }}(\bm{x})}\,d\bm{x}\leq \sqrt{\epsilon}+\frac{1}{m}\log\frac{N_{\b}\big(\mP(\mathcal{F}_{\theta}),\epsilon\big)}{\delta}.
\%
Set $\epsilon=1/m^2$. We have
\%
H^2\big({p}_{f_{\hat\theta}}(x,x't),p_{f_{\theta^* }}(x,x',t)\big)\leq \frac{2}{m}\log\frac{N_{\b}\big(\mP(\mathcal{F}_{\theta}),1/m^2\big)}{\delta}.
\%



\end{proof}
