\begin{align}& \textrm{if } \lambda\in L^p(\calF_t,\P) \textrm{ and } Y\in L^p(\calF_{t+1},\P), \textrm{ then } \rho_t(Y+\lambda)=\rho_t(Y)-\lambda, \\ & \textrm{if } Y,\widetilde{Y}\in L^p(\calF_{t+1},\P) \textrm{ and } Y\leq \widetilde{Y}, \textrm{ then } \rho_t(Y)\geq \rho_t(\widetilde{Y}), \\ & \rho_t(0)=0.\end{align}
