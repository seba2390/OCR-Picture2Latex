\begin{equation}\begin{split} \left(\chi_A -\bar{\chi}_{\bar{A}}\right)=& c\left(\int_0^{\bar{z}_A + \eta}d\tilde{z}\; \frac{1}{H(\tilde{z})} -\int_0^{\bar{z}_A}d\tilde{z}\; \frac{1}{\bar{H}(\tilde{z})} \right)\\ =& c\left(\int_0^{\bar{z}_A}d\tilde{z}\; \left(\frac{1}{H(\tilde{z})}-\frac{1}{\bar{H}(\tilde{z})}\right) + \int_{\bar{z}_A}^{\bar{z}_A + \eta}d\tilde{z}\frac{1}{H(\tilde{z})}\right)\\ \approx& \; \Delta \chi_{A} + c\frac{\eta}{H(\bar{z}_A)}\;, \end{split}\end{equation}
