\begin{align}& \text{ rank$\left(\sum_{i \notin S} U_S \cap U_i \right)$} \\ &= \text{ rank$\left(\sum_{i \in [k-2]\setminus A} U_A \cap V_1 \cap V_2 \cap U_i \right)$} \\ & = \text{ rank$\left(\sum_{i \in [k-2]\setminus A} U_A \cap V_1 \cap U_i \right)$} \\ & = \text{ rank$\left(\sum_{i \in [k-1]\setminus (A\sqcup \{k-1\})} U_{A \sqcup \{k-1\} } \cap U_i \right)$} \\ & = \sum_{ [k-1] \supseteq T \sqcup \{k-1\} \supsetneq A \sqcup \{k-1\} } (-1)^{(|A|+1)-(|T|+1)+1} \; \rank(U_T \cap V_1) \\ & \qquad \qquad \qquad \text{(by \autoref{CBPCriterionPID}, since $\{U_1, \dots U_{k-2}, V_1\}$ satisfies inclusion-exclusion),} \\ & = \sum_{[k-2] \supseteq T\supsetneq A } (-1)^{|A|-|T|+1} \; \rank(U_T \cap V_1) \\ & = \sum_{[k-2] \supseteq T\supsetneq A } (-1)^{|A|-|T|+1 } \; \rank(U_A \cap V_1 \cap V_2) \\ & = \sum_{[k] \supseteq T \sqcup \{k-1, k\} \supsetneq A \cup \{k-1, k\} } (-1)^{|A|-|T|+1 } \; \rank(U_A \cap V_1 \cap V_2) \\ & = \sum_{[k] \supseteq T' \supsetneq S } (-1)^{|A|-|T'|+1} \; \rank(U_{T'}).\end{align}
