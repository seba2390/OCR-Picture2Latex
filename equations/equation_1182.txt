\begin{align}& \textrm{if } \lambda\in L^p(\calF_t,\P) \textrm{ and } Y\in L^p(\calF_{t+1},\P), \textrm{ then } \phi_t(Y+\lambda)=\phi_t(Y)+\lambda, \\ & \textrm{if } Y,\widetilde{Y}\in L^p(\calF_{t+1},\P) \textrm{ and } Y\leq \widetilde{Y}, \textrm{ then } \phi_t(Y)\leq \phi_t(\widetilde{Y}), \\ & \phi_t(0)=0.\end{align}
