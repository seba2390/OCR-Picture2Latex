\begin{align}\notag\dfrac{1}{2^{n}}&\sum_{i_1,i_2,...,i_{n}=0}^{3}(w_{i_1,i_2,...,i_{n}}\sigma_{i_1,i_2,...,i_{n}}\otimes\sigma_{i_1,i_2,...,i_{n}})\sum_{i=0}^{d-1}\ket{ii} \\ \notag&=\dfrac{1}{2^{n}}\sum_{i_1,i_2,...,i_{n}=0}^{3}(w_{i_1,i_2,...,i_{n}}\mathbb{I}\otimes\sigma_{i_1,i_2,...,i_{n}}\sigma_{i_1,i_2,...,i_{n}}^T)\sum_{i=0}^{d-1}\ket{ii}\\ \notag&=\dfrac{1}{2^{n}}\sum_{i_1,i_2,...,i_{n}=0}^{3}(\mathbb{I}\otimes\mathbb{I})\sum_{i=0}^{d-1}\ket{ii}\\ \notag&=\dfrac{1}{2^{n}}(4^{n})\sum_{i=0}^{d-1}\ket{ii}\\ &=2^{n}\sum_{i=0}^{d-1}\ket{ii}.\end{align}
