\begin{align}\mathbb{E}{[S_{A}]} &= -\mathbb{E}\big[ \tr{\rho_{A}\log{\rho_{A}}}-\tr{\rho_{A}}\log{\tr{\rho_{A}}}\big]\\ & = -\partial_{n}\mathbb{E}\big[ \tr{\rho_{A}^{n}}-\left(\tr{\rho_{A}}\right)^{n} \big]\bigg|_{n\to 1}\\ &= -\partial_{n}\left[\frac{\mathbb{E}\tr{\rho^{n}}}{\mathbb{E}\left(\tr{\rho_{A}}\right)^{n}}\right]\bigg|_{n\to 1} \text{ (using $\mathbb{E}\tr{\rho_{A}}= 1$)}\\ &= \frac{1}{1-n}\log\frac{\mathbb{E}\tr{\rho^{n}}}{\mathbb{E}\left(\tr{\rho_{A}}\right)^{n}}\bigg|_{n\to 1} \\ &= \hat{S}^{(n\to 1)}_{A}.\end{align}
