\begin{align}&\nu(xF^iyF^j,zF^k)\nu(xF^j,yF^k)\\ &= \nu(xF^i(y)F^{i+j},zF^k)\nu(xF^i,yF^j)\\ &= s(z)\cdots s(F^{i+j-1}(z)) \cdot \alpha(xF^i(y),F^{i+j}(z)) \cdot s(y)\cdots s(F^{i-1}(y)) \cdot \alpha(x,F^i(y))\\ &= s(z)\cdots s(F^{i+j-1}(z))\cdot s(y)\cdots s(F^{j-1}(y))\cdot \alpha(xF^i(y),F^{i+j}(z)))\cdot \alpha(x,F^i(y))\\ \intertext{We now use the $2$-cocycle condition on $\alpha$} &= s(z)\cdots s(F^{i+j-1}(z))\cdot s(y)\cdots s(F^{j-1}(y))\cdot \alpha(x,F^i(y)F^{i+j}(z)))\cdot \alpha(F^i(y),F^{i+j}(z))\\ &= s(z)\cdots s(F^{j-1}(z)) \cdot \alpha(x,F^i(yF^j(z))) \cdot s(y)s(F^j(z))\cdots s(F^{i-1}(y))s(F^{i+j-1}(z)) \cdot \alpha(F^i(y),F^{i+j}(z))\\ \intertext{By assumption we have for any $a,b \in G$ that $\alpha(a,b)s(ab) = \alpha(F(a),F(b))s(a)s(b)$. Applying this repeatedly we get} &= s(z)\cdots s(F^{j-1}(z))\cdot \alpha(x,F^i(yF^j(z))) \cdot s(yF^j(z))\cdots s(F^{i-1}(yF^j(z))) \cdot \alpha(y,F^j(z))\\ &= s(yF^j(z))\cdots s(F^{i-1}(yF^j(z))) \cdot \alpha(x,F^i(yF^j(z))) \cdot s(z)\cdots s(F^{j-1}(z)) \cdot \alpha(y,F^j(z))\\ &= \nu(xF^i,yF^j(z)F^{j+k})\nu(yF^j,zF^k)\\ &= \nu(xF^i,yF^jzF^k)\nu(yF^j,zF^k).\end{align}
