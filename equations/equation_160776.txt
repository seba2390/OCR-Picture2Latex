\begin{align}\left(\frac{{\pi}^{\frac{n}{2}}}{\Gamma(\frac{n}{2}+1)}\right)^{-\frac{2}{n-1}} &= \pi^{-1}(1+O(\tfrac{1}{n}))\left(\Gamma\left(\frac{n}{2}+1\right)\right)^{\frac{2}{n-1}} \nonumber\\ &=\pi^{-1}(1+O(\tfrac{1}{n}))\left(\sqrt{2\pi\cdot\frac{n}{2}}\left(\frac{n}{2}\right)^{\frac{n}{2}}e^{-\frac{n}{2}}\right)^{\frac{2}{n-1}}\nonumber\\ &=(2\pi e)^{-1}(n+2\ln n+O(1)).\end{align}
