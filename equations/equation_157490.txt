\begin{align}KL[q(\bs{\theta}, \bs{z}) \| p(\bs{\theta}, \bs{z})] = &\mathbb{E}_{q(\bs{\theta}, \bs{z})} [\log{q (\bs{\theta}, \bs{z})} - \log{p(\bs{\theta},\bs{z})}]\nonumber\\ = &\mathbb{E}_{q(\bs{\theta},\bs{z})} [\log{q (\bs{\theta} | \bs{z})q(\bs{z})} - \log{p(\bs{\theta} | \bs{z})p(\bs{z})}]\nonumber\\ = &\mathbb{E}_{q(\bs{\theta},\bs{z})} [\log{q (\bs{\theta} | \bs{z})} - \log{p(\bs{\theta} | \bs{z})} + \log{q(\bs{z})-\log p(\bs{z})}]\nonumber\\ = & KL[q(\bs{\theta} | \bs{z}) || p(\bs{\theta} | \bs{z})] + KL[q(\bs{z}) || p(\bs{z})].\end{align}
