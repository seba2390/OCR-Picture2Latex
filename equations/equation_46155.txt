\begin{equation}\frac{\partial f}{\partial t}= \frac{\partial f}{\partial \varepsilon_{ij}}\frac{\partial \varepsilon_{ij}}{\partial t}+ \frac{\partial f}{\partial \chi}\frac{\partial \chi}{\partial t}+ \frac{\partial f}{\partial \dot{\chi}}\frac{\partial \dot{\chi}}{\partial t} + \frac{\partial f}{\partial \chi_{,i}}\frac{\partial \chi_{,i}}{\partial t}+ \frac{\partial f}{\partial T}\frac{\partial T}{\partial t}+ \frac{\partial f}{\partial T_{,i}}\frac{\partial T_{,i}}{\partial t} .\end{equation}
