    \begin{minipage}{.9\linewidth{}}\textbf{\corelang{} typing rules \ \ \ \fbox{\ensuremath{\Gamma \,\vdash\, t:A}}}\end{minipage}
    \\\\
    \begin{minipage}{.30\linewidth}
        \infrule[int]{
             \\% \\
        }{
            \emptyset \,\vdash\, n : \textsf{Int}
        }
    \end{minipage}
    \begin{minipage}{.30\linewidth}
        \infrule[var]{
            \\
        }{
            x:A \,\vdash\, x:A
        }
    \end{minipage}
    \begin{minipage}{.35\linewidth}
        \infrule[abs]{
            \Gamma, x:A \,\vdash\, t : B%\theta B
        }{
            \Gamma \,\vdash\, \lam{x}{t} : \ftype{A}{B}
        }
    \end{minipage}
    \\\\
    \begin{minipage}{.95\linewidth}
        \infrule[app]{
            \Gamma_1 \,\vdash\, t_1 : \ftype{A}{B}
            \andalso
            \Gamma_2 \,\vdash\, t_2 : A
        }{
            \Gamma_1 + \Gamma_2 \,\vdash\, \app{t_1}{t_2} : B
        }
    \end{minipage}
    \\\\
    \begin{minipage}{.95\linewidth}
        \infrule[let]{
            \Gamma_1 \,\vdash\, t_1 : \vertype{r}{A}
            \andalso
            \Gamma_2, x:\verctype{A}{r} \,\vdash\, t_2 : B
        }{
            \Gamma_1 + \Gamma_2 \,\vdash\, \clet{x}{t_1}{t_2} : B
        }
    \end{minipage}
    \\\\
    \begin{minipage}{.30\linewidth}
        \infrule[weak]{
            \Gamma \,\vdash\, t : A
        }{
            \Gamma, \verctype{\Delta}{0} \,\vdash\, t : A
        }
    \end{minipage}
    \begin{minipage}{.35\linewidth}
        \infrule[der]{
            \Gamma, x:A \,\vdash\, t : B
        }{
            \Gamma, x:\verctype{A}{1} \,\vdash\, t : B
        }
    \end{minipage}
    \begin{minipage}{.30\linewidth}
        \infrule[pr]{
            [\Gamma] \,\vdash\, t : A 
        }{
            r\cdot[\Gamma] \,\vdash\, [t] : \vertype{r}{A} 
        }
    \end{minipage}
    \\\\
    \begin{minipage}{.50\linewidth}
        \infrule[sub]{
            \Gamma,x:\verctype{A}{r}, \Gamma' \,\vdash\, t : B
            \andalso
            r \sqsubseteq s
        }{
            \Gamma,x:\verctype{A}{s}, \Gamma' \,\vdash\, t : B
        }
    \end{minipage}
    \begin{minipage}{.45\linewidth}
        \infrule[extr]{
            \Gamma \,\vdash\, t : \vertype{r}{A}
            \andalso
            l \in r
        }{
            \Gamma \,\vdash\, t.l : A
        }
    \end{minipage}
    \\\\
    \begin{minipage}{.95\linewidth}
        \infrule[ver]{
            [\Gamma_i] \,\vdash\, t_i : A
        }{
            \bigcup_i(\{l_i\}\cdot [\Gamma_i]) \,\vdash\, \{\overline{l=t}\,|\,l_i\} : \vertype{\{\overline{l}\}}{A}
        }
    \end{minipage}
    \\\\
    \begin{minipage}{.95\linewidth}
        \infrule[veri]{
            [\Gamma_i] \,\vdash\, t_i : A
        }{
            \bigcup_i(\{l_i\}\cdot [\Gamma_i]) \,\vdash\, \langle\overline{l=t}\,|\,l_i\rangle : A
        }
    \end{minipage}
\end{tabular}
