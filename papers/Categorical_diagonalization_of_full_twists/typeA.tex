\begin{document}



\begin{abstract} We conjecture that the complex of Soergel bimodules associated with the full twist braid is categorically diagonalizable, for any finite Coxeter group. This utilizes the theory of categorical diagonalization introduced in an earlier paper \cite{ElHog17a}. We prove our conjecture in type $A$, and as a result we obtain a categorification of the Young idempotents.
\end{abstract}


\title{Categorical diagonalization of full twists}
