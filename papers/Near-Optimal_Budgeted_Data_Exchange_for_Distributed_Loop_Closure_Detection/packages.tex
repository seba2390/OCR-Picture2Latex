	\newcommand\DoubleLine[5][4pt]{%
	  \path(#2)--(#3)coordinate[at start](h1)coordinate[at end](h2);
	  \draw[<-,line width=1mm,black] ($(h1)!#1!90:(h2)$)  to ["#4"]   ($(h2)!#1!-90:(h1)$);
	  \draw[->,line width=1mm,black] ($(h1)!#1!-90:(h2)$) to ["#5" '] ($(h2)!#1!90:(h1)$);
	}
