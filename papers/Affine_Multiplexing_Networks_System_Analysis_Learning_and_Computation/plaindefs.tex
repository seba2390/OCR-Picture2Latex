% allows the use of smallcaps within math mode
% https://tex.stackexchange.com/a/106117
\font\tensc=cmcsc10
\font\sevensc=cmcsc10 at 7pt
\font\fivesc=cmcsc10 at 5pt

\def\mathsc#1{{\mathchoice
  {\hbox{\tensc#1}}
  {\hbox{\tensc#1}}
  {\hbox{\sevensc#1}}
  {\hbox{\fivesc#1}}
}}

\def\ttbox#1{\mbox{\tt #1}}
\def\scbox#1{\mbox{\sc #1}}
\def\bfbox#1{\mbox{\bf #1}}
