% make \veca ... \vecz and \matA ... \matZ and \setA ... \setZ available in math mode ( https://tex.stackexchange.com/questions/45712/faster-way-of-writing-mathbf )
\makeatletter
\@tfor\next:=abcdefghijklmnopqrstuvwxyz\do{%
  \def\command@factory#1{%
    \expandafter\def\csname vec#1\endcsname{\mathbf{#1}}
  }
 \expandafter\command@factory\next
}

\@tfor\next:=ABCDEFGHIJKLMNOPQRSTUVWXYZ\do{%
  \def\command@factory#1{%
    \expandafter\def\csname mat#1\endcsname{\mathbf{#1}}
  }
 \expandafter\command@factory\next
}

\@tfor\next:=ABCDEFGHIJKLMNOPQRSTUVWXYZ\do{%
  \def\command@factory#1{%
    \expandafter\def\csname set#1\endcsname{\mathcal{#1}}
  }
 \expandafter\command@factory\next
}


% greek letters
\def\greekvectors#1{%
 \@for\next:=#1\do{%
    \def\X##1;{%
     \expandafter\def\csname mat##1\endcsname{\boldsymbol{\csname##1\endcsname}}
     }
  \expandafter\X\next;
  }
}
