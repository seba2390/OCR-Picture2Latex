\usepackage{tikz, marginnote} % margin notes with circles
\newcommand{\checkedby}[1]{
\ifdefined\CROSSCHECKS
  \marginnote{
    \begin{tikzpicture}
      \foreach \x [count=\xi] in {#1} {
         \node[shape=circle,inner sep=0mm,
         minimum size=2mm,
         fill=\csname \x color\endcsname] at (\xi*3mm,0) {};
       }
    \end{tikzpicture}
  }
\else
\fi
}
