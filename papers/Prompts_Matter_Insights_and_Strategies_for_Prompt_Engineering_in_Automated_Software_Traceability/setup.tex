% % Found a new way of commenting which is quite nice (JCH)
\usepackage{ifthen} 
\newboolean{showcomments}
\setboolean{showcomments}{true} % toggle to show or hide comments
\ifthenelse{\boolean{showcomments}}
   {
   
  \newcommand{\pinkcomment}[2]{
     \fcolorbox{gray}{magenta}{\bfseries\sffamily\scriptsize#1}
     {\sf\small$\blacktriangleright$\textit{#2}$\blacktriangleleft$}
   }
  
   \newcommand{\purplecomment}[2]{
     \fcolorbox{gray}{violet}{\bfseries\sffamily\scriptsize#1}
     {\sf\small$\blacktriangleright$\textit{#2}$\blacktriangleleft$}
   }
  
   \newcommand{\bluecomment}[2]{
     \fcolorbox{gray}{blue}{\bfseries\sffamily\scriptsize#1}
     {\sf\small$\blacktriangleright$\textit{#2}$\blacktriangleleft$}
   }
    
   \newcommand{\orangecomment}[2]{
     \fcolorbox{gray}{orange}{\bfseries\sffamily\scriptsize#1}
     {\sf\small$\blacktriangleright$\textit{#2}$\blacktriangleleft$}
   }
  
   \newcommand{\todocomment}[2]{
     \fcolorbox{gray}{yellow}{\bfseries\sffamily\scriptsize#1}
     {\sf\small$\blacktriangleright$\textit{#2}$\blacktriangleleft$}
   }

    \newcommand{\redcomment}[2]{
     \fcolorbox{gray}{red}{\bfseries\sffamily\scriptsize#1}
     {\sf\small$\blacktriangleright$\textit{#2}$\blacktriangleleft$}
   }
   
   \newcommand{\version}{\emph{\scriptsize$-$working$-$}}
}
