\begin{algorithm}[H]
    \small
    \caption{BetaZero action progressive widening.}
    \label{alg:betazero-action-pw}
    \begin{algorithmic}[1]
        \Function{ActionSelection$(b)$}{}
            \State $\tilde{b} \leftarrow \textproc{Representation}(b)$
            \If {$| A(b) | \le kN(s)^\alpha$}
                \State $a \sim P_\theta(\tilde{b}, \cdot)$ \GrayComment{prioritize actions from policy network}
                \State $N(b,a), Q(s,a) \leftarrow N_0(b,a), Q_0(b,a)$
                \State $A(b) \leftarrow A(b) \cup \{a\}$
            \EndIf
            \State \Return ${\displaystyle\argmax_{a \in A(s)}}\ Q(b,a) + c\left(P_\theta(\tilde{b},a)\frac{\sqrt{N(b)}}{1 + N(b,a)}\right)$ \GrayComment{PUCT}
        \EndFunction
    \end{algorithmic}
\end{algorithm}
