\begin{algorithm}[H]
    \small
    \caption{State progressive widening.}
    \label{alg:mcts-state-pw}
    \begin{algorithmic}[1]
        \Function{StateExpansion$(s,a)$}{}
            \If {$|S(s,a)| \le kN(s,a)^\alpha$ \textbf{or} $S(s,a) = \emptyset$}
                \State $s' \sim T(\cdot \mid s, a)$
                \State $S(s,a) \leftarrow S(s,a) \cup \{s'\}$
            \Else
                \State $s' \sim S(s,a)$
            \EndIf
            \State $r \leftarrow R(s, a, s')$ % TODO: s' or s????
            \State \Return $s', r$
        \EndFunction
    \end{algorithmic}
\end{algorithm}
